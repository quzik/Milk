let customers = []; // Populate Month Selector ONCE const monthSelect = document.getElementById("month"); const now = new Date(); for(let i = 0; i < 6; i++) { let d = new Date(now.getFullYear(), now.getMonth() - i, 1); let val = d.toISOString().slice(0, 7); // "2024-05" let opt = document.createElement("option"); opt.value = val; opt.textContent = val; monthSelect.appendChild(opt); } async function login() { const user = document.getElementById("user"); const pass = document.getElementById("pass"); try { const res = await fetch('/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, // ← REQUIRED body: JSON.stringify({ username: user.value, password: pass.value }) }); if (!res.ok) throw new Error(`HTTP ${res.status}`); loadCustomers(); } catch (err) { alert("Login Failed: " + err.message); console.error(err); } } async function loadCustomers() { try { const res = await fetch('/customers'); if (!res.ok) throw new Error(`HTTP ${res.status}`); customers = await res.json(); const savedRes = await fetch(`/load?month=${encodeURIComponent(monthSelect.value)}`); if (!savedRes.ok) throw new Error(`HTTP ${savedRes.status}`); const savedData = await savedRes.json(); renderTable(savedData); } catch (err) { alert("Failed to load data: " + err.message); console.error(err); } } function renderTable(savedData = []) { const tbody = document.getElementById("tbody"); tbody.innerHTML = customers.map(c => { const saved = savedData.find(s => parseInt(s.customer_id) === parseInt(c.id)) || { qty: 1, rate: 50 }; return ` ${c.name} 0 `; }).join(""); calc(); } function calc() { document.querySelectorAll("#tbody tr").forEach(tr => { const q = parseFloat(tr.querySelector(".qty").value) || 0; const r = parseFloat(tr.querySelector(".rate").value) || 0; const isDelivered = tr.querySelector(".delivered").checked; tr.querySelector(".total").innerText = isDelivered ? (q * r).toFixed(2) : "0.00"; }); } async function save() { try { const rows = [...document.querySelectorAll("#tbody tr")].map(tr => ({ customer_id: parseInt(tr.dataset.id), qty: parseFloat(tr.querySelector(".qty").value) || 0, rate: parseFloat(tr.querySelector(".rate").value) || 0, delivered: tr.querySelector(".delivered").checked // ← boolean, not array })); const res = await fetch('/save', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ month: monthSelect.value, rows }) }); if (!res.ok) throw new Error(`HTTP ${res.status}`); alert("Data Saved Successfully!"); } catch (err) { alert("Save Failed: " + err.message); console.error(err); } }

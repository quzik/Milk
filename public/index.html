let customers = [], chart; // Month dropdown const months = ["Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dec"]; const monthEl = document.getElementById("month"); months.forEach((m,i)=>{ let o=document.createElement("option"); o.value=i+1; o.text=m; monthEl.appendChild(o); }); monthEl.value = new Date().getMonth()+1; // ---------- LOGIN ---------- async function login() { const userBtn = document.getElementById("user"); const passBtn = document.getElementById("pass"); const msg = document.getElementById("msg"); const res = await fetch("/login",{ method:"POST", headers:{"Content-Type":"application/json"}, body:JSON.stringify({username:userBtn.value, password:passBtn.value}) }); const d = await res.json(); if(!res.ok){ msg.className="error"; msg.innerText=d.error; return; } // Refresh page on success to trigger the (async()) check below location.reload(); } // ---------- LOGOUT ---------- async function logout(){ await fetch("/logout"); // Method defaults to GET in your worker for /logout location.reload(); } // ---------- CUSTOMERS ---------- async function addCustomer(){ const cname = document.getElementById("cname"); const res = await fetch("/customer",{ method:"POST", headers:{"Content-Type":"application/json"}, body:JSON.stringify({name:cname.value}) }); if(res.ok){ cname.value=""; loadCustomers(); } } async function loadCustomers(){ const res = await fetch("/customers"); if(res.ok) { customers = await res.json(); render(); } } // ---------- TABLE ---------- function render(){ const tbody = document.getElementById("tbody"); tbody.innerHTML = customers.map(c=>` ${c.name} 50.00 `).join(""); calc(); } function calc(){ document.querySelectorAll("#tbody tr").forEach(tr=>{ let q = parseFloat(tr.querySelector(".qty").value) || 0; let r = parseFloat(tr.querySelector(".rate").value) || 0; tr.querySelector(".total").innerText = (q * r).toFixed(2); }); } // ---------- SAVE ---------- async function save(){ const smsg = document.getElementById("smsg"); let rows = [...document.querySelectorAll("#tbody tr")].map(tr=>({ customer_id: tr.dataset.id, qty: tr.querySelector(".qty").value, rate: tr.querySelector(".rate").value, days: [true] })); await fetch("/save",{ method:"POST", headers:{"Content-Type":"application/json"}, body:JSON.stringify({month: monthEl.value, rows}) }); smsg.innerText = "Saved!"; loadChart(); } // ---------- LOAD ENTRIES ---------- async function loadEntries(){ const res = await fetch("/load?month=" + monthEl.value); const data = await res.json(); document.querySelectorAll("#tbody tr").forEach(tr=>{ const e = data.find(x => x.customer_id == tr.dataset.id); if(e){ tr.querySelector(".qty").value = e.qty; tr.querySelector(".rate").value = e.rate; } }); calc(); } // ---------- CHART ---------- async function loadChart(){ const res = await fetch("/analytics"); const d = await res.json(); const ctx = document.getElementById("chartCanvas").getContext("2d"); if(chart) chart.destroy(); chart = new Chart(ctx, { type: "bar", data: { labels: d.map(x => months[parseInt(x.month) - 1]), datasets: [{label: "Revenue", data: d.map(x => x.revenue), backgroundColor: '#27ae60'}] } }); } // ---------- AUTO INIT ---------- (async ()=>{ const res = await fetch("/customers"); if(res.ok){ document.getElementById("login").style.display = "none"; document.getElementById("app").style.display = "block"; customers = await res.json(); render(); loadChart(); } })();
